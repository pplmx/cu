# Fuzz Testing CMake Configuration
# Enable with: cmake -B build -DNOVA_BUILD_FUZZ_TESTS=ON

option(NOVA_BUILD_FUZZ_TESTS "Build fuzz testing targets" OFF)

if(NOT NOVA_BUILD_FUZZ_TESTS)
    return()
endif()

# Check for libFuzzer (typically provided by compiler)
set(FUZZER_LIB "")
if(CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    set(FUZZER_LIB "Fuzzer")
elseif(CMAKE_CUDA_COMPILER_ID STREQUAL "NVIDIA")
    # CUDA uses clang-based driver on Clang toolchains
    if(CMAKE_CUDA_COMPILER MATCHES "clang")
        set(FUZZER_LIB "Fuzzer")
    endif()
endif()

if(NOT FUZZER_LIB)
    message(STATUS "Fuzz testing requires Clang/LLVM. Skipping fuzz targets.")
    return()
endif()

find_library(FUZZER_LIBRARY ${FUZZER_LIB} PATHS /usr/lib /usr/local/lib)
if(NOT FUZZER_LIBRARY)
    message(STATUS "libFuzzer not found. Install with: apt install libfuzzer-dev")
    return()
endif()

message(STATUS "Building fuzz testing targets with libFuzzer")

# Memory Pool Fuzz Target
add_executable(memory_pool_fuzz
    fuzz/memory_pool_fuzz.cpp
)
target_link_libraries(memory_pool_fuzz PRIVATE
    cuda_impl
    CUDA::cudart
    ${FUZZER_LIBRARY}
)
target_include_directories(memory_pool_fuzz PRIVATE
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/tests/fuzz
)
target_compile_options(memory_pool_fuzz PRIVATE -fsanitize=fuzzer)

add_custom_target(fuzz_memory_pool
    COMMAND memory_pool_fuzz $<TARGET_FILE:memory_pool_fuzz> ${CMAKE_SOURCE_DIR}/tests/fuzz/corpus/memory_pool -max_total_time=60
    DEPENDS memory_pool_fuzz
    COMMENT "Running memory pool fuzzing tests"
    VERBATIM
)
