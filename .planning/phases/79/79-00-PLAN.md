## ELL Format Specification

```cpp
template<typename T>
class SparseMatrixELL {
public:
    // Factory method
    static SparseMatrixELL<T> FromCSR(const SparseMatrixCSR<T>& csr);

    // Accessors
    int num_rows() const;
    int num_cols() const;
    int nnz() const;           // Actual non-zeros
    int padded_nnz() const;    // Storage with padding
    int max_nnz_per_row() const;
    const T* values() const;
    const int* col_indices() const;
    const int* row_offsets() const;  // Padded row offsets

private:
    std::vector<T> values_;
    std::vector<int> col_indices_;
    int num_rows_ = 0;
    int num_cols_ = 0;
    int max_nnz_per_row_ = 0;
};
```

## SELL Format Specification

```cpp
template<typename T>
class SparseMatrixSELL {
public:
    // Factory method with configurable slice height
    static SparseMatrixSELL<T> FromCSR(const SparseMatrixCSR<T>& csr, int slice_height = 32);

    // Accessors
    int num_rows() const;
    int num_cols() const;
    int nnz() const;
    int padded_nnz() const;
    int slice_height() const;
    const T* values() const;
    const int* col_indices() const;
    const int* slice_ptr() const;  // Offsets for each slice

private:
    std::vector<T> values_;
    std::vector<int> col_indices_;
    std::vector<int> slice_ptr_;
    int num_rows_ = 0;
    int num_cols_ = 0;
    int slice_height_ = 32;
};
```
