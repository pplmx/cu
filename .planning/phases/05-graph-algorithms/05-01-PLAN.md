   ```cpp
           CSRGraph(int num_vertices, int num_edges);
           ~CSRGraph();
           
           // Allocate device memory and copy data
           void upload();
           
           // Copy results back from device
           void download();
           
           // Free all memory (host and device)
           void clear();
           
           // Get vertex count
           int vertices() const { return num_vertices; }
           
           // Get edge count
           int edges() const { return num_edges; }
           
           // Get degree of vertex v

               return row_offsets[v + 1] - row_offsets[v];
   ```

   ```cpp
   #include "cuda/graph/csr_graph.h"
   #include "cuda/device/error.h"
   #include "cuda/algo/kernel_launcher.h"
   #include <algorithm>
   #include <numeric>
   
   namespace cuda::graph {
       CSRGraph::CSRGraph() : num_vertices(0), num_edges(0),
           row_offsets(nullptr), columns(nullptr), weights(nullptr),
           d_row_offsets(nullptr), d_columns(nullptr), d_weights(nullptr) {}
       
       CSRGraph::CSRGraph(int nv, int ne) : num_vertices(nv), num_edges(ne) {
           row_offsets = new int[nv + 1];
           columns = new int[ne];
           weights = new float[ne];
           
           CUDA_CHECK(cudaMalloc(&d_row_offsets, (nv + 1) * sizeof(int)));
           CUDA_CHECK(cudaMalloc(&d_columns, ne * sizeof(int)));
           CUDA_CHECK(cudaMalloc(&d_weights, ne * sizeof(float)));
       }
       
       CSRGraph::~CSRGraph() {
           clear();
       }
       
       void CSRGraph::clear() {
           delete[] row_offsets;
           delete[] columns;
           delete[] weights;
           
           if (d_row_offsets) cudaFree(d_row_offsets);
           if (d_columns) cudaFree(d_columns);
           if (d_weights) cudaFree(d_weights);
           
           row_offsets = columns = nullptr;
           weights = nullptr;
           d_row_offsets = d_columns = d_weights = nullptr;
           num_vertices = num_edges = 0;
       }
       
       void CSRGraph::upload() {
           CUDA_CHECK(cudaMemcpy(d_row_offsets, row_offsets, 
               (num_vertices + 1) * sizeof(int), cudaMemcpyHostToDevice));
           CUDA_CHECK(cudaMemcpy(d_columns, columns, 
               num_edges * sizeof(int), cudaMemcpyHostToDevice));
           CUDA_CHECK(cudaMemcpy(d_weights, weights, 
               num_edges * sizeof(float), cudaMemcpyHostToDevice));
       }
       
       void CSRGraph::download() {
           CUDA_CHECK(cudaMemcpy(row_offsets, d_row_offsets, 
               (num_vertices + 1) * sizeof(int), cudaMemcpyDeviceToHost));
           CUDA_CHECK(cudaMemcpy(columns, d_columns, 
               num_edges * sizeof(int), cudaMemcpyDeviceToHost));
           CUDA_CHECK(cudaMemcpy(weights, d_weights, 
               num_edges * sizeof(float), cudaMemcpyDeviceToHost));
       }
       
       size_t CSRGraph::memory_usage() const {
           size_t host = (num_vertices + 1) * sizeof(int) + 
                         num_edges * sizeof(int) + 
                         num_edges * sizeof(float);
           size_t device = host;  // Same size on device
           return host + device;
       }
   }
   ```

2. **Create CSR from Edge List** (GRAPH-04):
   ```cpp
   namespace cuda::graph {
       std::unique_ptr<CSRGraph> create_csr_from_edges(
           const int* src_vertices,
           const int* dst_vertices,
           const float* edge_weights,
           int num_vertices,
           int num_edges
       ) {
           auto graph = std::make_unique<CSRGraph>(num_vertices, num_edges);
           
           // Count outgoing edges per vertex
           std::fill(graph->row_offsets, graph->row_offsets + num_vertices + 1, 0);
           for (int i = 0; i < num_edges; ++i) {
               graph->row_offsets[src_vertices[i]]++;
           }
           
           // Compute prefix sum for row_offsets
           std::partial_sum(graph->row_offsets, graph->row_offsets + num_vertices + 1, 
                            graph->row_offsets);
           
           // Fill columns and weights
           std::vector<int> current_pos(num_vertices, 0);
           for (int i = 0; i < num_edges; ++i) {
               int src = src_vertices[i];
               int dst = dst_vertices[i];
               float w = edge_weights ? edge_weights[i] : 1.0f;
               
               int pos = graph->row_offsets[src] + current_pos[src]++;
               graph->columns[pos] = dst;
               graph->weights[pos] = w;
           }
           
           // Upload to device
           graph->upload();
           
           return graph;
       }
   }
   ```

3. **Create CSR from Adjacency List**:
   ```cpp
   namespace cuda::graph {
       std::unique_ptr<CSRGraph> create_csr_from_adjacency(
           const std::vector<std::vector<int>>& adjacency,
           bool weighted
       ) {
           int num_vertices = (int)adjacency.size();
           int num_edges = 0;
           for (const auto& neighbors : adjacency) {
               num_edges += (int)neighbors.size();
           }
           
           auto graph = std::make_unique<CSRGraph>(num_vertices, num_edges);
           
           // Fill row_offsets
           graph->row_offsets[0] = 0;
           for (int v = 0; v < num_vertices; ++v) {
               graph->row_offsets[v + 1] = graph->row_offsets[v] + adjacency[v].size();
           }
           
           // Fill columns
           int pos = 0;
           for (int v = 0; v < num_vertices; ++v) {
               for (int neighbor : adjacency[v]) {
                   graph->columns[pos] = neighbor;
                   graph->weights[pos] = 1.0f;
                   pos++;
               }
           }
           
           graph->upload();
           return graph;
       }
   }
   ```

4. **CSR Matrix-Vector Multiplication**:
   ```cpp
   namespace cuda::graph {
       // Sparse matrix-vector multiplication kernel
       __global__ void csr_mv_kernel(
           const int* row_offsets,
           const int* columns,
           const float* weights,
           const float* x,
           float* y,
           int num_vertices
       ) {
           int v = blockIdx.x * blockDim.x + threadIdx.x;
           if (v >= num_vertices) return;
           
           float sum = 0.0f;
           int start = row_offsets[v];
           int end = row_offsets[v + 1];
           
           for (int i = start; i < end; ++i) {
               int neighbor = columns[i];
               sum += weights[i] * x[neighbor];
           }
           
           y[v] = sum;
       }
       
       void csr_mv(
           const CSRGraph& graph,
           const float* x,
           float* y,
           cudaStream_t stream
       ) {
           int block = 256;
           int grid = (graph.num_vertices + block - 1) / block;
           
           csr_mv_kernel<<<grid, block, 0, stream>>>(
               graph.d_row_offsets,
               graph.d_columns,
               graph.d_weights,
               x, y, graph.num_vertices
           );
           CUDA_CHECK(cudaGetLastError());
       }
   }
   ```

   ```cpp
   namespace cuda::graph {
       bool validate_csr(const CSRGraph& graph) {
           // Check row_offsets are monotonically increasing
           for (int i = 0; i < graph.num_vertices; ++i) {
               if (graph.row_offsets[i + 1] < graph.row_offsets[i]) {
                   return false;
               }
           }
           
           // Check total edges matches
           if (graph.row_offsets[graph.num_vertices] != graph.num_edges) {
               return false;
           }
           
           // Check columns are valid vertex indices
           for (int i = 0; i < graph.num_edges; ++i) {
               if (graph.columns[i] < 0 || graph.columns[i] >= graph.num_vertices) {
                   return false;
               }
           }
           
           return true;
       }
   }
   ```

   ```cpp
   namespace cuda::graph {
       // Mark frontier vertices and their neighbors
       // GRAPH-01: Compute distance from source to all reachable vertices
       // GRAPH-02: Mark unreachable vertices as -1 (never visited)
       __global__ void bfs_frontier_kernel(
           const int* __restrict__ row_offsets,
           const int* __restrict__ columns,
           const int* __restrict__ distances,
           bool* __restrict__ visited,
           int* __restrict__ next_frontier,
           int* __restrict__ next_frontier_count,
           int current_level,
           int num_vertices
       ) {
           int v = blockIdx.x * blockDim.x + threadIdx.x;
           if (v >= num_vertices) return;
           
           // Check if this vertex is in current frontier
           if (distances[v] == current_level) {
               int start = row_offsets[v];
               int end = row_offsets[v + 1];
               
               for (int i = start; i < end; ++i) {
                   int neighbor = columns[i];
                   
                   // Use atomic compare-and-swap to mark unvisited neighbors
                   if (atomicExch((int*)&visited[neighbor], 1) == 0) {
                       // Successfully marked as visited
                       int pos = atomicAdd(next_frontier_count, 1);
                       next_frontier[pos] = neighbor;
                   }
               }
           }
       }
       
       // Update distances for next frontier
       __global__ void bfs_update_distances_kernel(
           const int* __restrict__ next_frontier,
           int next_frontier_count,
           int next_level,
           int* __restrict__ distances
       ) {
           int idx = blockIdx.x * blockDim.x + threadIdx.x;
           if (idx >= next_frontier_count) return;
           
           int v = next_frontier[idx];
           distances[v] = next_level;
       }
       
       // Count visited vertices
       __global__ void bfs_count_visited_kernel(
           const bool* __restrict__ visited,
           int* __restrict__ visited_count,
           int num_vertices
       ) {
           int idx = blockIdx.x * blockDim.x + threadIdx.x;
           if (idx >= num_vertices) return;
           
           if (visited[idx]) {
               atomicAdd(visited_count, 1);
           }
       }
   ```

   ```cpp
           result.upload();

           CUDA_CHECK(cudaMalloc(&d_next_frontier, graph.num_vertices * sizeof(int)));
           CUDA_CHECK(cudaMalloc(&d_frontier_count, sizeof(int)));
           CUDA_CHECK(cudaMalloc(&d_next_frontier_count, sizeof(int)));
           
           // Initialize: source is the first frontier
           int* h_frontier = new int[graph.num_vertices];
           h_frontier[0] = source;
           CUDA_CHECK(cudaMemcpy(d_current_frontier, h_frontier, sizeof(int), cudaMemcpyHostToDevice));
           CUDA_CHECK(cudaMemcpy(d_frontier_count, &graph.num_vertices, sizeof(int), cudaMemcpyHostToDevice)); // Use temp
           int current_frontier_size = 1;
           
           int current_level = 0;
           int max_iterations = graph.num_vertices; // Safety limit
           
           int block = 256;
           
           while (current_frontier_size > 0 && current_level < max_iterations) {
               // Reset next frontier count
               CUDA_CHECK(cudaMemset(d_next_frontier_count, 0, sizeof(int)));
               
               // Process current frontier
               int grid = (graph.num_vertices + block - 1) / block;
               bfs_frontier_kernel<<<grid, block, 0, stream>>>(
                   graph.d_row_offsets, graph.d_columns,
                   result.d_distances, result.d_visited,
                   d_next_frontier, d_next_frontier_count,
                   current_level, graph.num_vertices
               );
               CUDA_CHECK(cudaGetLastError());
               
               // Get next frontier size
               int next_frontier_size;
               CUDA_CHECK(cudaMemcpy(&next_frontier_size, d_next_frontier_count, 
                   sizeof(int), cudaMemcpyDeviceToHost));
               
               if (next_frontier_size == 0) break;
               
               // Update distances for next frontier
               grid = (next_frontier_size + block - 1) / block;
               bfs_update_distances_kernel<<<grid, block, 0, stream>>>(
                   d_next_frontier, next_frontier_size, current_level + 1, result.d_distances
               );
               CUDA_CHECK(cudaGetLastError());
               
               // Swap frontiers
               std::swap(d_current_frontier, d_next_frontier);
               current_frontier_size = next_frontier_size;
               current_level++;
           }
           
           // Download results
           result.download();
           
           // Cleanup
           delete[] h_frontier;
           cudaFree(d_current_frontier);
           cudaFree(d_next_frontier);
   ```

2. **PageRank Power Iteration Kernel** (GRAPH-03):
   ```cpp
   namespace cuda::graph {
       // Compute out-degrees on device
       __global__ void pagerank_compute_outdegree_kernel(
           const int* __restrict__ row_offsets,
           float* __restrict__ out_degree,
           int num_vertices
       ) {
           int v = blockIdx.x * blockDim.x + threadIdx.x;
           if (v >= num_vertices) return;
           out_degree[v] = row_offsets[v + 1] - row_offsets[v];
           if (out_degree[v] == 0) out_degree[v] = 1.0f; // Handle sinks
       }
       
       // PageRank iteration: new_rank = (1-d)/n + d * sum(out_neighbors)
       // GRAPH-03: PageRank converges to 1e-6 tolerance within 50 iterations
       __global__ void pagerank_iteration_kernel(
           const int* __restrict__ row_offsets,
           const int* __restrict__ columns,
           const float* __restrict__ out_degree,
           const float* __restrict__ old_ranks,
           float* __restrict__ new_ranks,
           float damping,
           int num_vertices
       ) {
           int v = blockIdx.x * blockDim.x + threadIdx.x;
           if (v >= num_vertices) return;
           
           // Sum contributions from incoming edges
           // For each edge u -> v, add damping * rank[u] / out_degree[u]
           float sum = 0.0f;
           
           // We need to iterate through all vertices to find incoming edges
           // This is inefficient but correct; a more sophisticated approach
           // would use transpose representation
           for (int u = 0; u < num_vertices; ++u) {
               int start = row_offsets[u];
               int end = row_offsets[u + 1];
               
               for (int i = start; i < end; ++i) {
                   if (columns[i] == v) {
                       sum += old_ranks[u] / out_degree[u];
                       break;
                   }
               }
           }
           
           // Teleportation contribution (1 - damping) / n
           float teleport = (1.0f - damping) / num_vertices;
           
           new_ranks[v] = teleport + damping * sum;
       }
       
       // Optimized PageRank using transpose CSR
       __global__ void pagerank_iteration_transpose_kernel(
           const int* __restrict__ row_offsets,
           const int* __restrict__ columns,
           const float* __restrict__ out_degree,
           const float* __restrict__ old_ranks,
           float* __restrict__ new_ranks,
           float damping,
           int num_vertices
       ) {
           int v = blockIdx.x * blockDim.x + threadIdx.x;
           if (v >= num_vertices) return;
           
           // Sum contributions from incoming edges (same as above)
           float sum = 0.0f;
           for (int u = 0; u < num_vertices; ++u) {
               int start = row_offsets[u];
               int end = row_offsets[u + 1];
               
               for (int i = start; i < end; ++i) {
                   if (columns[i] == v) {
                       sum += old_ranks[u] / out_degree[u];
                   }
               }
           }
           
           float teleport = (1.0f - damping) / num_vertices;
           new_ranks[v] = teleport + damping * sum;
       }
       
       // Compute L1 norm of rank difference
       __global__ void pagerank_error_kernel(
           const float* __restrict__ old_ranks,
           const float* __restrict__ new_ranks,
           float* __restrict__ errors,
           int num_vertices
       ) {
           int v = blockIdx.x * blockDim.x + threadIdx.x;
           if (v >= num_vertices) return;
           errors[v] = fabsf(new_ranks[v] - old_ranks[v]);
       }
       
       __global__ void pagerank_reduce_error_kernel(
           const float* __restrict__ errors,
           float* __restrict__ total_error,
           int num_vertices
       ) {
           extern __shared__ float sdata[];
           int idx = blockIdx.x * blockDim.x + threadIdx.x;
           int tid = threadIdx.x;
           
           sdata[tid] = (idx < num_vertices) ? errors[idx] : 0.0f;
           __syncthreads();
           
           for (int s = blockDim.x / 2; s > 0; s >>= 1) {
               if (tid < s) {
                   sdata[tid] += sdata[tid + s];
               }
               __syncthreads();
           }
           
           if (tid == 0) {
               atomicAdd(total_error, sdata[0]);
           }
       }
   }
   ```

   ```cpp
   namespace cuda::graph {
       PageRankResult pagerank(
           const CSRGraph& graph,
           const PageRankOptions& options,
           cudaStream_t stream
       ) {
           PageRankResult result(graph.num_vertices);
           result.damping = options.damping;
           result.tolerance = options.tolerance;
           
           float* d_out_degree;
           float* d_new_ranks;
           float* d_errors;
           float* d_total_error;
           
           CUDA_CHECK(cudaMalloc(&d_out_degree, graph.num_vertices * sizeof(float)));
           CUDA_CHECK(cudaMalloc(&d_new_ranks, graph.num_vertices * sizeof(float)));
           CUDA_CHECK(cudaMalloc(&d_errors, graph.num_vertices * sizeof(float)));
           CUDA_CHECK(cudaMalloc(&d_total_error, sizeof(float)));
           
           // Compute out-degrees
           int block = 256;
           int grid = (graph.num_vertices + block - 1) / block;
           pagerank_compute_outdegree_kernel<<<grid, block, 0, stream>>>(
               graph.d_row_offsets, d_out_degree, graph.num_vertices
           );
           
           // Initialize ranks
           float init_rank = 1.0f / graph.num_vertices;
           CUDA_CHECK(cudaMemcpy(result.d_ranks, &init_rank, sizeof(float), 
               cudaMemcpyHostToDevice));
           CUDA_CHECK(cudaMemset(result.d_ranks + 1, 0, (graph.num_vertices - 1) * sizeof(float)));
           
           // Power iteration
           float* d_old_ranks;
           CUDA_CHECK(cudaMalloc(&d_old_ranks, graph.num_vertices * sizeof(float)));
           
           bool converged = false;
           for (int iter = 0; iter < options.max_iterations; ++iter) {
               // Copy current ranks to old
               CUDA_CHECK(cudaMemcpy(d_old_ranks, result.d_ranks, 
                   graph.num_vertices * sizeof(float), cudaMemcpyDeviceToDevice));
               
               // PageRank iteration
               pagerank_iteration_kernel<<<grid, block, 0, stream>>>(
                   graph.d_row_offsets, graph.d_columns, d_out_degree,
                   d_old_ranks, result.d_ranks, options.damping, graph.num_vertices
               );
               CUDA_CHECK(cudaGetLastError());
               
               // Check convergence
               CUDA_CHECK(cudaMemset(d_total_error, 0, sizeof(float)));
               pagerank_error_kernel<<<grid, block, 0, stream>>>(
                   d_old_ranks, result.d_ranks, d_errors, graph.num_vertices
               );
               pagerank_reduce_error_kernel<<<grid, block, block * sizeof(float), 0, stream>>>(
                   d_errors, d_total_error, graph.num_vertices
               );
               
               float total_error;
               CUDA_CHECK(cudaMemcpy(&total_error, d_total_error, sizeof(float), 
                   cudaMemcpyDeviceToHost));
               
               result.iterations = iter + 1;
               
               // GRAPH-03: Converge to 1e-6 tolerance within 50 iterations
               if (total_error < options.tolerance) {
                   converged = true;
                   break;
               }
           }
           
           // Download results
           result.download();
           
           // Normalize if requested
           if (options.normalize && converged) {
               result.normalize();
           }
           
           // Cleanup
           cudaFree(d_out_degree);
           cudaFree(d_new_ranks);
           cudaFree(d_errors);
           cudaFree(d_total_error);
           cudaFree(d_old_ranks);
           
           return result;
       }
       
       PageRankResult pagerank_with_stats(
           const CSRGraph& graph,
           PageRankStats& stats,

           cudaStream_t stream
       ) {
           auto result = pagerank(graph, options, stream);
           stats.iterations = result.iterations;
           stats.converged = result.iterations < options.max_iterations;
           stats.final_error = result.tolerance;
           return result;
       }
   ```
