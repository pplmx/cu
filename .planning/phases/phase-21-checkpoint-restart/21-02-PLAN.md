```cpp
#pragma once

#include <string>
#include <map>
#include <vector>
#include <variant>
#include <cuda_runtime.h>

namespace nova {

// Tensor representation for serialization
struct SerializedTensor {
    std::string name;
    std::vector<int64_t> shape;
    int dtype;  // cudaDataType_t
    size_t size_bytes;
    std::vector<char> data;
};

// Checkpoint manifest
struct CheckpointManifest {
    int version = 1;
    int step;
    int64_t timestamp;
    std::string model_hash;
    std::string optimizer_hash;
    std::vector<SerializedTensor> model_tensors;
    std::vector<SerializedTensor> optimizer_tensors;
    bool has_rng_state;
    std::vector<char> rng_state;
};

class StateSerializer {
public:
    static StateSerializer& instance();
    
    // Serialization
    CheckpointManifest serialize_model(
        const std::map<std::string, std::vector<float>>& params
    );
    
    CheckpointManifest serialize_optimizer(
        const std::map<std::string, OptimizerState>& states
    );
    
    std::vector<char> serialize_rng_state(cudaStream_t stream);
    
    // Deserialization
    std::map<std::string, std::vector<float>> deserialize_model(
        const CheckpointManifest& manifest
    );
    
    void deserialize_optimizer(
        const CheckpointManifest& manifest,
        std::map<std::string, OptimizerState>& states
    );
    
    void deserialize_rng_state(
        const std::vector<char>& state,
        cudaStream_t stream
    );
    
    // File I/O
    void save_to_file(
        const CheckpointManifest& model_state,
        const CheckpointManifest& optimizer_state,
        const std::vector<char>& rng_state,
        const std::string& path
    );
    
    struct LoadedCheckpoint {
        CheckpointManifest model;
        CheckpointManifest optimizer;
        std::vector<char> rng;
    };
    
    LoadedCheckpoint load_from_file(const std::string& path);
    
    // Validation
    bool validate_checkpoint(const std::string& path);
    std::string compute_hash(const std::vector<char>& data);
    
private:
    StateSerializer() = default;
    ~StateSerializer() = default;
    
    void write_tensor(std::ostream& os, const SerializedTensor& tensor);
    SerializedTensor read_tensor(std::istream& is);
    void write_string(std::ostream& os, const std::string& str);
    std::string read_string(std::istream& is);
};
```

### Implementation: `src/cuda/checkpoint/state_serializer.cpp`

Key implementation details:

1. **Binary format with header**:
   - Magic bytes: "CKPT"
   - Version: uint32_t
   - Tensor count: uint32_t
   - For each tensor: name, shape, dtype, size, data

2. **Safetensors-inspired format**:
   - No pickle (security)
   - Memory mapping for large tensors
   - Shape/dtype stored separately

3. **Optimizer state handling**:
   - Different optimizers have different state
   - Store type discriminator
   - Adam: {exp_avg, exp_avg_sq}

4. **RNG state**:
   - Use CUDA runtime API: cudaGet_rng_state
   - Store as raw bytes

### RNG State Handling

```cpp
// Save RNG state
size_t rng_state_size;
cudaStreamGetRNGStateSize(stream, &rng_state_size);  // pseudo-code
std::vector<char> rng_state(rng_state_size);
cudaGetRNGState(stream, rng_state.data());

// Restore RNG state
cudaSetRNGState(stream, rng_state.data(), rng_state.size());
```
