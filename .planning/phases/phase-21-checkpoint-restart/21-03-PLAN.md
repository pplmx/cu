### Implementation: `src/cuda/checkpoint/checkpoint_manager.cpp`

Key implementation details:

1. **Incremental Checkpoint Logic**:

```cpp
// Track tensor checksums
std::map<std::string, std::string> tensor_checksums_;

// On each checkpoint:
// 1. Compute current checksums
// 2. Compare with last checkpoint
// 3. Only save changed tensors
// 4. Store manifest of which tensors are in incremental

void CheckpointManager::save_incremental_checkpoint(int step) {
    auto current_checksums = compute_checksums();
    
    // Load last full checkpoint manifest
    auto last_manifest = load_manifest(last_checkpoint_path_);
    
    std::vector<std::string> changed_tensors;
    for (const auto& [name, checksum] : current_checksums) {
        if (!last_manifest.checksums.contains(checksum) ||
            last_manifest.checksums[name] != checksum) {
            changed_tensors.push_back(name);
        }
    }
    
    // If >50% changed, do full checkpoint instead
    if (changed_tensors.size() > total_tensors_ * 0.5) {
        save_full_checkpoint(step);
    } else {
        save_incremental_checkpoint(step, changed_tensors);
    }
}
```

```cpp
class ErrorSignal {
public:

    void emit_error();  // Call on detected error
};

// Integration with existing error handling
void on_nccl_error() {
    ErrorSignal::instance().emit_error();
}

void on_cuda_error(cudaError_t err) {
    if (is_recoverable_error(err)) {
        ErrorSignal::instance().emit_error();
    }
}
```

3. **Atomic Write with Rename**:

```cpp
void safe_atomic_write(const std::string& path, 
                       const void* data, 
                       size_t size) {
    std::string temp = path + ".tmp." + std::to_string(getpid());
    write_file(temp, data, size);
    rename(temp.c_str(), path.c_str());  // Atomic on POSIX
}
```

### Coordination for Multi-Rank

For multi-node checkpoints, coordinate via NCCL:

```cpp
// Rank 0 writes, others wait
void coordinated_checkpoint() {
    if (is_rank_0()) {
        save_checkpoint(step);
    }
    
    // Barrier to ensure all ranks see consistent state
    mesh_barrier();
    
    // If rank 0 succeeded, others load
    if (!is_rank_0()) {
        load_checkpoint(step);
    }
    
    mesh_barrier();
}
```
