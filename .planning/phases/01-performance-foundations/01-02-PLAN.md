   ```cpp
       const char* operation_name = nullptr;

       int device_id = -1;
       std::string extra;  // Additional context
   };
   ```

2. **Add enhanced exception**:
   ```cpp
   class CudaExceptionWithContext : public CudaException {
   public:
       CudaExceptionWithContext(cudaError_t err, const char* file, int line,
                                const OperationContext& ctx);
       const OperationContext& context() const { return context_; }
   private:
       OperationContext context_;
   };
   ```

3. **Update format_error to include context**:
   - Format: `{file}:{line} - CUDA error: {error_string} [{operation_name}] dims={dimensions} device={device_id}`

4. **Add helper macro for operation context**:
   ```cpp
   #define CUDA_CONTEXT(op, dims, device) \
       cuda::device::OperationContext{#op, dims, device}
   ```

5. **Add input validation macro**:
   ```cpp
   #define CUDA_VALIDATE_SIZE(size, max_size, operation) \
       do { \
           if ((size) > (max_size)) { \
               throw CudaExceptionWithContext(cudaErrorLaunchFailure, __FILE__, __LINE__, \
                   CUDA_CONTEXT(operation, size, 0)); \
           } \
       } while(0)
   ```

6. **DO NOT modify existing CudaException behavior** - add new types only
</action>
