### 2. Add NOVA_USE_CCACHE option to CMakeLists.txt

```cmake
option(NOVA_USE_CCACHE "Enable ccache for faster rebuilds" OFF)

if(NOVA_USE_CCACHE)
    find_program(CCACHE_PROGRAM ccache)
    if(CCACHE_PROGRAM)
        message(STATUS "ccache found: ${CCACHE_PROGRAM}")
        set(CMAKE_CUDA_COMPILER_LAUNCHER ${CCACHE_PROGRAM})
        set(CMAKE_CXX_COMPILER_LAUNCHER ${CCACHE_PROGRAM})
    else()
        message(WARNING "ccache not found, but NOVA_USE_CCACHE=ON")
    endif()
endif()
```
