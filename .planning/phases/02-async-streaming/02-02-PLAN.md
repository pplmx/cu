1. **Direction enum** and **copy context struct**:
   ```cpp
   namespace cuda::async {
       enum class CopyDirection {
           HostToDevice,
           DeviceToHost,
           DeviceToDevice,
           HostToHost
       };
       
       struct CopyContext {
           const void* src = nullptr;
           void* dst = nullptr;
           size_t bytes = 0;
           CopyDirection direction = CopyDirection::HostToDevice;
           cudaStream_t stream = nullptr;
           const char* label = nullptr;  // For profiling
       };
   }
   ```

2. **AsyncCopy namespace** with high-level functions:
   ```cpp
   namespace cuda::async {
       // Basic async copy with stream
       void async_copy(void* dst, const void* src, size_t bytes,
                       cudaStream_t stream);
       
       // Copy with direction enum
       void async_copy(void* dst, const void* src, size_t bytes,
                       CopyDirection dir, cudaStream_t stream);
       
       // Copy with context (for profiling/debugging)
       void async_copy(const CopyContext& ctx);
       
       // Host to Device convenience
       inline void async_copy_h2d(void* device_ptr, const void* host_ptr,
                                  size_t bytes, cudaStream_t stream);
       
       // Device to Host convenience
       inline void async_copy_d2h(void* host_ptr, const void* device_ptr,
                                  size_t bytes, cudaStream_t stream);
       
       // Device to Device convenience
       inline void async_copy_d2d(void* dst_device, const void* src_device,
                                  size_t bytes, cudaStream_t stream);
       
       // Batched copy from pinned memory
       template <typename T>
       void async_copy_from_pinned(T* device_ptr, const PinnedMemory<T>& pinned,
                                   size_t offset, size_t count, cudaStream_t stream);
       
       // Batched copy to pinned memory
       template <typename T>
       void async_copy_to_pinned(PinnedMemory<T>& pinned, const T* device_ptr,
                                 size_t offset, size_t count, cudaStream_t stream);
       
       // 2D copy (for non-contiguous data)
       void async_copy_2d(void* dst, size_t dpitch,
                          const void* src, size_t spitch,
                          size_t width, size_t height,
                          CopyDirection dir, cudaStream_t stream);
   }
   ```

   - 2D copy uses `cudaMemcpy2DAsync`

5. **Namespace**: `cuda::async`

   };

2. **Add stream-aware allocation API**:
   ```cpp
   // Allocate with optional stream association
   void* allocate(size_t bytes, int stream_id = -1);
   
   // Track allocations per stream
   std::unordered_map<int, size_t> get_allocations_by_stream() const;
   
   // Synchronize allocations for a specific stream

   ```

3. **Enhance PoolMetrics**:
   ```cpp
   struct PoolMetrics {
       size_t hits = 0;

       size_t peak_allocated_bytes = 0;
       int num_active_streams = 0;
   };
   ```

4. **Add graceful failure option**:
   ```cpp
   void set_throw_on_failure(bool throw_on_failure);
   bool get_throw_on_failure() const;
   ```

1. **Update allocate() method**:
   - Accept optional stream_id parameter
   - Store stream_id in allocation_map
   - Update get_metrics() to track peak_allocated_bytes and num_active_streams

2. **Implement get_allocations_by_stream()**:
   - Iterate allocation_map and group by stream_id
   - Return map of stream_id -> total bytes allocated

3. **Implement synchronize_stream()**:
