   ```cpp
   struct BenchmarkOptions {
       int warmup_iterations = 3;    // BMCH-01: warm-up runs
       int measurement_iterations = 10;
       bool compute_throughput = false;
       size_t data_size_bytes = 0;   // For throughput calculation
       double tolerance_percent = 10.0;  // For regression detection (BMCH-04)
   };
   ```

3. **Benchmark class**:

   ```cpp
   class Benchmark {
   public:
       explicit Benchmark(std::string name, BenchmarkOptions options = {});
       ~Benchmark();

       // Run a kernel and measure execution time
       template<typename Kernel, typename... Args>
       BenchmarkResult run(Kernel&& kernel, Args&&... args);

       // Get/set baseline for regression detection
       void set_baseline(const BenchmarkResult& baseline);
       bool compare_to_baseline(const BenchmarkResult& current) const;
       std::string format_regression_report(const BenchmarkResult& current) const;

   private:
       std::string name_;
       BenchmarkOptions options_;
       std::optional<BenchmarkResult> baseline_;
       std::vector<double> measurements_;

       void warmup();
       BenchmarkResult compute_statistics();
   };
   ```

4. **Throughput helper function**:

   ```cpp
   double compute_throughput_gbps(size_t bytes, double elapsed_ms);
   ```

5. **Implementation details**:
   - Use `cudaEvent_t` for precise timing
   - warmup() executes the kernel options.warmup_iterations times without measuring
   - compute_statistics() calculates mean, stddev, min, max
   - throughput_gbps = (data_size_bytes / 1e9) / (elapsed_ms / 1000.0)
   - compare_to_baseline() returns true if current is within tolerance_percent of baseline mean

6. **Include guard**: `#pragma once`
7. **Dependencies**: `<cuda_runtime.h>`, `<vector>`, `<string>`, `<optional>`
8. **Namespace**: `cuda::benchmark`
9. **Create directory**: `include/cuda/benchmark/`

```cpp
// BMCH-03: Throughput measurement
TEST_F(ThroughputTest, MemoryCopyThroughputInGBps) {
    constexpr size_t N = 100 * 1024 * 1024;  // 100 MB
    constexpr double expected_min_gbps = 100.0;  // Minimum expected for modern GPU

    // Simple memory copy kernel
    auto kernel = [N]() {
        void* d_src;
        void* d_dst;
        cudaMalloc(&d_src, N);
        cudaMalloc(&d_dst, N);
        cudaMemcpy(d_dst, d_src, N, cudaMemcpyDeviceToDevice);
        cudaFree(d_src);
        cudaFree(d_dst);
    };

    Benchmark bench("memory_throughput", BenchmarkOptions{
        .compute_throughput = true,
        .data_size_bytes = N
    });

    auto result = bench.run(kernel);

    // BMCH-03: Throughput reported in GB/s
    EXPECT_GT(result.throughput_gbps, 0.0);
    EXPECT_GE(result.throughput_gbps, expected_min_gbps);
}

TEST_F(ThroughputTest, ThroughputCalculationIsCorrect) {
    // Test helper function directly
    double gbps = compute_throughput_gbps(1e9, 1.0);  // 1 GB in 1 ms
    EXPECT_DOUBLE_EQ(gbps, 1000.0);  // 1000 GB/s

    gbps = compute_throughput_gbps(1e9, 1000.0);  // 1 GB in 1 second
    EXPECT_DOUBLE_EQ(gbps, 1.0);  // 1 GB/s
}
```

```cpp
// BMCH-04: Regression detection
TEST_F(RegressionTest, NoRegressionWhenPerformanceIsWithinTolerance) {
    Benchmark bench("regression_test");

    BenchmarkResult baseline{/* mean: 10ms */ .mean_ms = 10.0, .stddev_ms = 1.0, .samples = 10};
    bench.set_baseline(baseline);

    // Current: 10% slower (still within 10% tolerance)
    BenchmarkResult current{/* mean: 11ms */ .mean_ms = 11.0, .stddev_ms = 1.0, .samples = 10};

    EXPECT_TRUE(bench.compare_to_baseline(current));  // Should pass
}

TEST_F(RegressionTest, RegressionDetectedWhenPerformanceDegrades) {
    Benchmark bench("regression_test", BenchmarkOptions{.tolerance_percent = 10.0});

    BenchmarkResult baseline{/* mean: 10ms */ .mean_ms = 10.0, .stddev_ms = 1.0, .samples = 10};
    bench.set_baseline(baseline);

    // Current: 15% slower (exceeds 10% tolerance)
    BenchmarkResult current{/* mean: 11.5ms */ .mean_ms = 11.5, .stddev_ms = 1.0, .samples = 10};

    EXPECT_FALSE(bench.compare_to_baseline(current));  // Should detect regression
}

TEST_F(RegressionTest, FormatReportShowsDeltaPercent) {
    Benchmark bench("regression_test");
    BenchmarkResult baseline{/* mean: 100ms */ .mean_ms = 100.0, .stddev_ms = 5.0, .samples = 10};
    bench.set_baseline(baseline);

    BenchmarkResult current{/* mean: 110ms */ .mean_ms = 110.0, .stddev_ms = 5.0, .samples = 10};

    std::string report = bench.format_regression_report(current);

    // Should contain delta information
    EXPECT_TRUE(report.find("+10") != std::string::npos || 
                report.find("10%") != std::string::npos);
}
```
