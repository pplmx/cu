   ```cpp
   namespace cuda::performance {
       struct DeviceProperties {
           int compute_capability_major;
           int compute_capability_minor;
           size_t global_memory_bytes;
           size_t shared_memory_per_block;
           int max_threads_per_block;
           int multiprocessor_count;
           size_t memory_bandwidth_gbps;
       };

       DeviceProperties get_device_properties(int device_id = 0);
       int get_optimal_block_size(int device_id = 0);  // Returns 128/256/512/1024 based on device

   }
   ```

     - 7.x (Volta): 256-512 threads
     - 8.x (Ampere): 256-512 threads
     - 9.x (Ada): 512-1024 threads
   - Add `namespace cuda::performance` directory if needed

3. **Include guard**: `#pragma once`
4. **Dependencies**: `<cuda_runtime.h>` only
5. **Namespace**: `cuda::performance`

   ```cpp
   }
   ```

2. **Add static helper function for device-aware grid calculation**:

   ```cpp
   [[nodiscard]] inline dim3 calc_grid_auto(size_t n, int device_id = 0) {
       dim3 block{static_cast<unsigned int>(cuda::performance::get_optimal_block_size()), 1, 1};
       return dim3{static_cast<unsigned int>((n + block.x - 1) / block.x), 1, 1};
   }
   ```
