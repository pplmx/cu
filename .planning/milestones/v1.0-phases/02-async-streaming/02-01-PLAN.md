       struct StreamConfig {

           // Pre-create streams for given priorities
           void initialize(int num_streams, int max_priority = 0);

           // Synchronize all managed streams

           // Query all streams

           // Get stream with highest priority (lowest number)

       private:
           std::vector<std::unique_ptr<cuda::stream::Stream>> streams_;

       };

1. **PinnedMemory template class** - RAII wrapper for page-locked memory:

   ```cpp
   namespace cuda::async {
       template <typename T>
       class PinnedMemory {
       public:
           PinnedMemory() = default;
           explicit PinnedMemory(size_t count);
           ~PinnedMemory();

           PinnedMemory(const PinnedMemory&) = delete;
           PinnedMemory& operator=(const PinnedMemory&) = delete;

           PinnedMemory(PinnedMemory&& other) noexcept;
           PinnedMemory& operator=(PinnedMemory&& other) noexcept;

           T* get() { return data_; }
           const T* get() const { return data_; }
           T* data() { return data_; }
           const T* data() const { return data_; }
           size_t size() const { return size_; }
           size_t size_bytes() const { return size_ * sizeof(T); }
           explicit operator bool() const { return data_ != nullptr; }

           void reset(size_t count);
           T* release();

       private:
           T* data_ = nullptr;
           size_t size_ = 0;
       };

       // Factory functions
       template <typename T>
       [[nodiscard]] PinnedMemory<T> make_pinned(size_t count);
   }
   ```

2. **Specialized PinnedBuffer class** with copy helpers:

   ```cpp
   template <typename T>
   class PinnedBuffer : public PinnedMemory<T> {
   public:
       using PinnedMemory<T>::PinnedMemory;

       // Copy from host vector
       void copy_from(const std::vector<T>& host_data);

       // Copy to host vector
       void copy_to(std::vector<T>& host_data) const;

       // Async copy to device
       template <typename StreamT>
       void copy_to_device(T* device_ptr, StreamT stream);

       // Async copy from device
       template <typename StreamT>
       void copy_from_device(const T* device_ptr, StreamT stream);
   };
   ```

   - Support move semantics properly
